\block{UD Treebanks in SLA Research}{
    Advantages of UD treebanks in corpus-based Second Language Acquisition research:
    \vskip 0.5em
    \begin{itemize}
        \item uniform morphosyntactic annotation layer allowing for qualitative and quantitative \textbf{cross-lingual comparisons}:
        \begin{itemize}
            \item between standard and learner language
            \item between a learner's L1 and L2
            \item between different L2s
        \end{itemize}
        \item possibility to carry out \textbf{grammatical error retrieval and analysis} and \textbf{automatic feedback generation} without explicit error tagging e.g. if learner sentences are paired with corrections (\textbf{\textit{parallel L1-L2 treebank}})
        \item \textbf{semi-automatic annotation} with the help of the existing UD parsers
    \end{itemize}
}

\block{The SweLL-UD treebank}{
    \vskip 10pt
    \innerblock{Objectives} {
        \begin{itemize}
            \item end goal: building a \textbf{training-scale treebank} of L2 Swedish
            \item within the next few months: releasing a high-quality \textbf{500-sentence test set}
            \item at the training school: further experimenting with \textbf{annotation of L2 essays} and discussing \textbf{open questions}
        \end{itemize}
    }
}

\column{0.5}

\block{}{
    \vskip -20pt
    \innerblock{Data}{
        Source corpus: \textbf{SweLL-gold} (Swedish Learner Language corpus)
        \begin{itemize}
            \item \textbf{genre}: essays (various topics)
            \item \textbf{learners}: adult L2 Swedish students with various language backgrounds and proficiency levels
            \item \textbf{annotation}: manual correction, error tagging and pseudonymization
            \item \textbf{size}: 502 essays (5000+ parallel sentences)
            \item \textbf{license}: CLARIN-ID -PRIV \underline{-NORED} -BY (but the data can and has been redistributed as long as it is impossible to reconstruct full essays)
        \end{itemize}
    }
    \vskip 5pt
    \innerblock{Project status and plan}{
        \begin{enumerate}
            \item \textbf{preprocessing}: sentence pair and metadata extraction, CoNNL-U format conversion, shuffling and creation of train, development and test splits \checkmark
            \item \textbf{preliminary annotation experiments} with a small annotation team \checkmark
            \item \textbf{L1 (corrections) annotation}:
            \begin{itemize}
                \item automatic pre-annotation with a pretrained UDPipe model \checkmark
                \item manual validation with a larger annotation team including both \newline L1 and L2 Swedish speakers
            \end{itemize}
            \item \textbf{L2 (originals) annotation}:
            \begin{itemize}
                \item automatic pre-annotation with a custom (fine-tuned) model
                \item manual validation with the larger annotation team
            \end{itemize}
        \end{enumerate}
    }
    \vskip 5pt
    \innerblock{Open questions}{
        \begin{itemize}
            \item \textbf{tooling} - desiderata:
            \begin{itemize}
                \item tree visualization
                \item no installation required (as with Eevee)
                \item parallel L1-L2 editing (or at least viewing, as with STUnD)
            \end{itemize}
            \item \textbf{L2-specific guidelines}:
            \begin{itemize}
                \item when annotating an original learner sentence, to what extent should the syntactic annotation of its correction be taken into account?
                \item what is a good balance between the general principle of \textit{literal reading} and keeping the two trees as structurally similar as possible?
                \item should the existing L1-specific guidelines be applied when annotating constructionsn that the learners borrow from their L1?
            \end{itemize}
        \end{itemize}
    }
}
