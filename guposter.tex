\block{UD Treebanks in SLA Research}{
    Benefits of UD treebanks in corpus-based Second Language Acquisition research:
    \vskip 0.5em
    \begin{itemize}
        \item uniform morphosyntactic annotation layer allowing for \textbf{cross-lingual \newline comparisons}:
        \begin{itemize}
            \item between standard and learner language
            \item between a learner's L1 and L2
            \item between different L2s
        \end{itemize}
        \item possibility to carry out \textbf{grammatical error analysis} without explicit error tagging e.g. if learner sentences are paired with corrections (\textit{parallel L1-L2 treebank})
        \item \textbf{semi-automatic annotation} through existing parsers
    \end{itemize}
}

\block{The SweLL-UD treebank}{
    \innerblock{Objectives} {
        \begin{itemize}
            \item ultimate goal: building a \textbf{training-scale treebank} of L2 Swedish
            \item within the next few months: releasing a high-quality \textbf{500-sentence test set}
            \item at the training school: discussing \textbf{L2-specific guidelines} and \textbf{tooling}
        \end{itemize}
    }
    \innerblock{Data}{
        Source corpus: \textbf{SweLL-gold} (Swedish Learner Language corpus)
        \begin{itemize}
            \item \textbf{text type}: short essays
            \item \textbf{learners}: adult L2 Swedish students with various language backgrounds and proficiency levels
            \item \textbf{annotation}: manual correction + error tagging + pseudonymization
            \item \textbf{size}: 502 essays, \textasciitilde 5000 parallel sentences
            \item \textbf{license}: CLARIN-ID -PRIV \underline{-NORED} -BY (but the data can be and has been redistributed as long as it is not possible to reconstruct full essays)
        \end{itemize}
    }
    \innerblock{Project status and plan}{
        \begin{enumerate}
            \item \textbf{preprocessing}: sentence pair and metadata extraction, conversion to CoNNL-U, shuffling and creation of a train, development and test split \checkmark
            \item \textbf{preliminary annotation experiments} with a small annotation team \checkmark
            \item \textbf{L1 (corrections) annotation}:
            \begin{enumerate}
                \item automatic pre-annotation with an off-the-shelf parser\checkmark
                \item manual validation with a larger annotation team (hopefully mixed L1 and L2 speakers, students and researchers)
            \end{enumerate}
            \item \textbf{L2 (originals) annotation}:
            \begin{enumerate}
                \item automatic pre-annotation with parser fine-tuned on the manually validated corrections $\uparrow$
                \item manual validation with the same annotation team
            \end{enumerate}
        \end{enumerate}
    }
}
